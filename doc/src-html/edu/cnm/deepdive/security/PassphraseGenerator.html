004package edu.cnm.deepdive.security;
005
006import java.security.SecureRandom;
007import java.util.ArrayList;
008import java.util.Enumeration;
009import java.util.Random;
010import java.util.ResourceBundle;
011
012/**
013 * Creates a passphrase of specified length from a predetermined word list.
014 * 
015 * @author Sky Link
016 */
017public class PassphraseGenerator {
018  
019  /** The word list used if none other is specified */
020  public static final String DEFAULT_WORD_LIST = "resources/wordlist";
021  /** A space is used if no other delimiter is specified */
022  public static final String DEFAULT_DELIMITER = " ";
023  /** The default number of words to be used in the passphrase*/
024  public static final int DEFAULT_LENGTH = 6;
025  
026  private String wordList = DEFAULT_WORD_LIST;
027  private String delimiter = DEFAULT_DELIMITER;
028  private int length = DEFAULT_LENGTH;
029  private Random rng = null;
030  private ArrayList<String> pool = null;
031  
032  /**
033   * Creates instance. Same functionality as default constructor.
034   */
035  public PassphraseGenerator() {
036    super();
037  }
038  
039  /**
040   * Creates a pool of words by adding them to an array list from the word list file.
041   */
042  protected void setupPool() {
043    ResourceBundle bundle = ResourceBundle.getBundle(wordList);
044    pool = new ArrayList<>();
045    Enumeration<String> keyEnum = bundle.getKeys();
046    while (keyEnum.hasMoreElements()) {
047      String key = keyEnum.nextElement();
048      String word = bundle.getString(key);
049      pool.add(word);
050    }
051  }
052  
053  /**
054   * Creates a secure random number generator
055   */
056  protected void setupRng() {
057    rng = new SecureRandom();
058  }
059  
060  /**
061   * Generates a passphrase from a pool of words
062   * 
063   * @return The passphrase created
064   */
065  public String generate() {
066    if (pool == null) {
067      setupPool();
068    }
069    if (rng == null) {
070      setupRng();
071    }
072    StringBuilder builder = new StringBuilder();
073    String word = pool.get(rng.nextInt(pool.size()));
074    builder.append(word);
075    for (int i = 0; i < length - 1; i++) {
076      word = pool.get(rng.nextInt(pool.size()));
077      builder.append(delimiter);
078      builder.append(word);
079    }
080    return builder.toString().trim();
081  }
082  
083  /**
084   * @return the wordList
085   */
086  public String getWordList() {
087    return wordList;
088  }
089  /**
090   * @param wordList the wordList to set
091   */
092  public void setWordList(String wordList) {
093    this.wordList = wordList;
094  }
095  /**
096   * @return the delimiter
097   */
098  public String getDelimiter() {
099    return delimiter;
100  }
101  /**
102   * @param delimiter the delimiter to set
103   */
104  public void setDelimiter(String delimiter) {
105    this.delimiter = delimiter;
106  }
107  /**
108   * @return the length
109   */
110  public int getLength() {
111    return length;
112  }
113  /**
114   * @param length the length to set
115   */
116  public void setLength(int length) {
117    this.length = length;
118  }
119
120}
